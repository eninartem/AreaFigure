create table times 
(
	id 			bigint unique generated by default as identity,
	time_name	varchar(100)
);

insert into times (time_name) values ('Утро'), ('День'), ('Вечер'), ('Август'), ('Декабрь'), ('Понедельник'), ('Яблоко'), ('Груша');

create table categories (
	id 				bigint unique generated by default as identity,
	category_name 	varchar(100)
);

insert into categories (category_name) values ('Время суток'), ('Месяц'), ('День недели');

create table times_categories (
	time_id bigint REFERENCES times(id),
	category_id bigint REFERENCES categories(id),
	PRIMARY KEY (time_id, category_id)
);

insert into times_categories values (1, 1), (2, 1), (3, 1), (4, 2), (5, 2), (6, 3);

SELECT t.time_name, c.category_name
FROM times t
LEFT JOIN times_categories tc
	ON t.id = tc.time_id
LEFT JOIN categories c
	ON tc.category_id = c.id;
